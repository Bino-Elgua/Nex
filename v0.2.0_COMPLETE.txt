  import { MultiAgentDispatcher } from "./multi-agent-dispatcher";

Usage Examples:
  // Guard resolution
  const resolver = new GuardConflictResolver();
  const result = resolver.resolve([allowGuard, denyGuard]);
  // → { detected: true, winner: denyGuard, reason: "DENY wins..." }

  // Multi-agent dispatching
  const dispatcher = new MultiAgentDispatcher(100);
  const id = await dispatcher.spawnAgent({role, goal, graph});
  await dispatcher.waitForCompletion([id]);
  const stats = dispatcher.getStats();
  // → { totalAgents: 100, completed: 100, failed: 0 }

  // Rewrite stability testing
  const validator = new RewriteValidator();
  const results = await validator.stressTest(1000);
  // → { total: 1000, passed: 1000, passRate: 100, avgMs: 1.2 }
